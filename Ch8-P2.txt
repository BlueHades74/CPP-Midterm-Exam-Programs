#include <iostream>

using namespace std;

void randomizeArray(int arr[5][5]){
    for (int i = 0; i < 5; i++){
        for (int j = 0; j < 5; j++){
            arr[i][j] = rand() % 10 + 1;
        }
    }
}

void printArray(int arr[5][5]){
    for (int i = 0; i < 5; i++){
        cout << "[";
        for (int j = 0; j < 5; j++){
            cout << arr[i][j] << " ";
        }
        cout << "]\n";
    }
}

void largestInRow(int arr[5][5]){
    
    int largest = 0, row, column;
    
    for (int i = 0; i < 5; i++){
        for (int j = 0; j < 5; j++){
            if (largest <= arr[i][j]){
                largest = arr[i][j];
                row = i + 1;
                column = j + 1;
            }
        }
        cout << "The largest in row " << row << " is " << largest 
            << " at column " << column << endl << endl;
        largest = 0;
        row = i;
        column = 0;
    }
}

int main() {
    
    int arr[5][5];
    
    randomizeArray(arr);
    
    printArray(arr);
    cout << "\n\n";
    largestInRow(arr);
    
    return 0;
}
