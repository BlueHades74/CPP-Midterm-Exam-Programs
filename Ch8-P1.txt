#include <iostream>

using namespace std;

void randomizeArray(int arr[10]){
    for (int i = 0; i < 10; i++){
        arr[i] = rand() % 10 + 1;
    }
}

int main() {
    
    int arr[10];
    
    randomizeArray(arr);
    
    for (int i = 0; i < 10; i++){
        cout << arr[i] << endl;
    }
    
    return 0;
}
