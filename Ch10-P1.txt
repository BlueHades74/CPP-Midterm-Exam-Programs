#include <iostream>
#include <string>

using namespace std;

class demon{
    string name;
    
    public:
        demon(){
            cout << "RELEASE ME AT ONCE!!!";
        }
        demon(string givenName){
            cout << "Hello, my name is " << name
                << ". RELEASE ME AT ONCE!!!";
            name = givenName;
        }
        
        void printName(){
            cout << "\n\n" << name;
        }
};

int main() {
    demon demon1;
    cout << endl;
    demon demon2("Dave");
    demon2.printName();
    
    
    return 0;
}
