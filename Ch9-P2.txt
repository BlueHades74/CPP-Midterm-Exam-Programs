#include <iostream>
#include <string>

using namespace std;

struct friendGroup{
    string name;
    int age;
};

int main() {
    friendGroup friends[5];
    bool canGroupDrink = true;
    
    for (int i = 0; i < 5; i++){
        cout << "Enter the name and age of a friend: ";
        cin >> friends[i].name >> friends[i].age;
        if (friends[i].age < 21) canGroupDrink = false;
    }
    
    if (canGroupDrink) cout << "YOU GUYS CAN GO TO THE CLUB!!!";
    else cout << "If you guys are cool, you'll invest in fake ID's";
    
    return 0;
}
