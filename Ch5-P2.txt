#include <iostream>

using namespace std;

int main() {
    
    int sum = 0, num;
    
    while(true){
        cout << "Enter a number greater than 0: ";
        cin >> num;
        
        if (num < 1) break;
        
        sum += num;
    }
    
    for (int i = 0; i < 10; i++){
        for (int j = i + 1; j < 20; j++){
            cout << "*";
        }
        cout << endl;
    }
        
    cout << "The sum of all the numbers is " << sum << endl;
    
    for (int i = 10; i > 0; i--){
        for (int j = i; j < 20; j++){
            cout << "*";
        }
        cout << endl;
    }
    
    return 0;
}
